\chapter{Энергетический расчет на оптических элементах}

\section{Исходные данные}

Для расчета плотностей энергии и мощности на оптических элементах используем следующие параметры:

\begin{itemize}
    \item Средняя мощность лазера: $P = 40$ Вт 
    \item Частота повторения импульсов: $f = 100$ кГц
    \item Длительность импульса: $\tau = 1$ мкс 
    \item Энергия в импульсе: $E_p = P/f = 0.4$ мДж
\end{itemize}

\section{Входная линза телескопа}

\begin{itemize}
    \item Диаметр пучка: 3.6 мм
    \item Площадь пятна: $S = \pi \cdot (1.8)^2 = 10.18$ мм$^2$ = 0.1018 см$^2$
    \item Плотность энергии: $\varepsilon = E_p/S = 3.93$ мДж/см$^2$
    \item Плотность мощности: $q = \varepsilon/\tau = 3.93$ кВт/см$^2$
\end{itemize}

\section{Первое зеркало}
\begin{itemize}
    \item Расстояние от перетяжки после телескопа до зеркала: $s = 0.2$ м
    \item Диаметр пучка: $D = 2\omega_{0,\text{нов}} \sqrt{1 + \left(\frac{s}{z_{R,\text{нов}}}\right)^2} = 17.54$ мм
    \item Площадь пятна: $A = \pi \cdot (8.77)^2 = 241.9$ мм$^2$ = 2.419 см$^2$
    \item Энергия на зеркале: $E = E_p \cdot 0.94 = 0.376$ мДж
    \item Плотность энергии: $\varepsilon = 0.376/2.419 = 0.155$ мДж/см$^2$
    \item Плотность мощности: $q = 0.155/1 \cdot 10^{-6} = 155$ кВт/см$^2$
\end{itemize}

\section{Фокусирующая линза}
\begin{itemize}
    \item Диаметр пучка: 17.55 мм
    \item Площадь пятна: 2.419 см$^2$ 
    \item Плотность энергии: $\varepsilon = 0.372/2.419 = 0.154$ мДж/см$^2$  
    \item Плотность мощности: $q = 0.154/1 \cdot 10^{-6} = 154$ кВт/см$^2$
\end{itemize}
